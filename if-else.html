گو با مثال: If/Else

دسته‌بندی با if و else در Go خیلی ساده است.

 
package main
 
import "fmt"
 
func main() {

این یک مثال ساده است.

 
    if 7%2 == 0 {
        fmt.Println("۷ زوج است")
    } else {
        fmt.Println("۷ فرد است")
    }

می‌تونین یه دستور if بدون else داشته باشین.

 
    if 8%4 == 0 {
        fmt.Println("۸ بر ۴ بخش‌پذیر است")
    }

عملگرهای منطقی مثل && و || اغلب تو شرایط مفید هستن.

 
    if 8%2 == 0 || 7%2 == 0 {
        fmt.Println("یا ۸ یا ۷ زوج است")
    }

یک دستور می‌تونه قبل از شرط‌ها بیاد؛ هر متغیری که تو این دستور تعریف بشه، تو همه بخش‌های بعدی در دسترس خواهد بود.

 
    if num := 9; num < 0 {
        fmt.Println(num, "منفی است")
    } else if num < 10 {
        fmt.Println(num, "یک رقمی است")
    } else {
        fmt.Println(num, "چند رقمی است")
    }
}

توجه کنین که در Go نیاز به پرانتز دور شرایط نیست ولی آکولادها الزامی هستند.

 
$ go run if-else.go
۷ فرد است
۸ بر ۴ بخش‌پذیر است
یا ۸ یا ۷ زوج است
9 یک رقمی است

در Go هیچ if سه‌تایی وجود نداره، بنابراین حتی برای شرایط ساده هم باید از یک دستور کامل if استفاده کنید.
