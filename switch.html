می‌تونید از کاما برای جدا کردن چندین عبارت در یک دستور case استفاده کنید. در این مثال، ما از default هم استفاده می‌کنیم.

 
    switch time.Now().Weekday() {
    case time.Saturday, time.Sunday:
        fmt.Println("آخر هفته است")
    default:
        fmt.Println("روز کاری است")
    }

switch بدون عبارت یک روش جایگزین برای بیان منطق if/else است. اینجا همچنین نشون می‌دیم که عبارت‌های case می‌تونن غیر ثابت باشن.

 
    t := time.Now()
    switch {
    case t.Hour() < 12:
        fmt.Println("قبل از ظهر است")
    default:
        fmt.Println("بعد از ظهر است")
    }

یک تایپ switch، به جای مقادیر، تایپ‌ها رو مقایسه می‌کنه. می‌تونید از این برای کشف تایپ یک مقدار interface استفاده کنید. در این مثال، متغیر t نوع مطابق با بخشش رو خواهد داشت.

 
    whatAmI := func(i interface{}) {
        switch t := i.(type) {
        case bool:
            fmt.Println("من یک bool هستم")
        case int:
            fmt.Println("من یک int هستم")
        default:
            fmt.Printf("نوع %T رو نمی‌دونم\n", t)
        }
    }
    whatAmI(true)
    whatAmI(1)
    whatAmI("hey")
}
